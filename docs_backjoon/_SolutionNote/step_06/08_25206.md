### 정석 코드 제안
#### 정석 코드

```cpp
#include <iostream>
#include <string>
#include <unordered_map>

int main()
{
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);

    std::unordered_map<std::string, double> score = 
    {
        {"A+", 4.5}, {"A0", 4.0},
        {"B+", 3.5}, {"B0", 3.0},
        {"C+", 2.5}, {"C0", 2.0},
        {"D+", 1.5}, {"D0", 1.0},
        {"F",  0.0}
    };

    double sumWeighted = 0.0;
    double sumCredit = 0.0;

    for (int i = 0; i < 20; ++i)
    {
        std::string subject, grade;
        double credit;

        std::cin >> subject >> credit >> grade;

        if (grade == "P") continue;

        sumWeighted += credit * score[grade];
        sumCredit += credit;
    }

    std::cout << sumWeighted / sumCredit;
    return 0;
}
```
