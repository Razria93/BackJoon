### 정석 코드 제안
#### 정석 코드
```cpp
#include <iostream>
#include <vector>
#include <algorithm>

int main()
{
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);

    int n;
    std::cin >> n;

    std::vector<int> a;
    a.reserve(n);

    for (int i = 0; i < n; ++i)
    {
        int x;
        std::cin >> x;
        a.push_back(x);
    }

    std::sort(a.begin(), a.end());

    for (int i = 0; i < n; ++i)
    {
        std::cout << a[i] << '\n';
    }

    return 0;
}
```
