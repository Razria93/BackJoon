### 정석 코드 제안
#### 정석 코드
```cpp
#include <iostream>

int digitSum(int x)
{
    int sum = 0;
    while (x > 0)
    {
        sum += x % 10;
        x /= 10;
    }
    return sum;
}

int main()
{
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);

    int N;
    std::cin >> N;

    int answer = 0;

    for (int x = 1; x < N; ++x)
    {
        if (x + digitSum(x) == N)
        {
            answer = x;
            break; // 가장 작은 생성자
        }
    }

    std::cout << answer;
    return 0;
}
```
